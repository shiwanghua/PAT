// 4. 无重复字符的最长子串的长度
#include<string.h>
int lengthOfLongestSubstring(char* s){
    int n=strlen(s), ans=1, i=0, j=0;
    int p[128];
    memset(p,-1,128*sizeof(int));
    while(j<n){
        int ci=s[j];
        if(p[ci]<i){
            j++;
            if(j-i>ans){
                ans=j-i;
                if(ans==128)
                  break;
            }
        }else{ 
            i=p[ci]+1;
        }
        p[ci]=j;
    }
    return ans;
}
