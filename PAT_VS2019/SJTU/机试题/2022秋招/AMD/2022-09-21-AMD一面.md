bool dfs(const std::vector<int> &a, int ai, int sum, const int target)
{
    if (ai == a.size())
    {
        return sum == target ? true : false;
    }
    return dfs(a, ai + 1, sum + a[ai], target) | dfs(a, ai + 1, sum - a[ai], target);
}
