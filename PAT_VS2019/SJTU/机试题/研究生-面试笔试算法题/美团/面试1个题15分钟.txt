#include<vector>

class Solution {
public:
    /**
     * 
     * @param s string字符串 
     * @return int整型
     */
    int numDecodings(string s) {
        // write code here
        int n=s.length();
        if(n==0) return 0;
//         int a=1,b=1,c=1;
//         for(int i=2;i<n;i++){
//             //c+=b;
//             if(stoi(s.substr(i-1,2))<=26)
//                 c+=a;
//             a=b;
//             b=c;
//             //c=0;
//         }
//         return c;

        vector<int> dp(n+1,0);
        dp[0]=1;
        if(s[0]!='0')
            dp[1]=1;
        else return 0;

        for(int i=1;i<n;i++){
            if(s[i]!='0')
                dp[i+1]+=dp[i];
            if(stoi(s.substr(i-1,2))<=26)
                dp[i+1]+=dp[i-1];
        }
        return dp[n];
    }
};
