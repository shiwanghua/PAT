uint32_t broadcastTime(uint32_t n, vector<vector<uint32_t>> edges){

    priority_queue<vector<uint32_t>> p; // min heap (要加greater<>)
    vector<vector<pair<uint32_t,uint32_t>>> e(n);
    for(auto& i:edges){
        e[i[0]].emplace_back(i[1],i[2]);
        e[i[1]].emplace_back(i[0],i[2]);
    }
    for(auto& i:e[0]){
        p.push({});
    }
}
